```cpp
enum class DirtyBit : uint32_t {
    Cpu = 1u << 0, Mem = 1u << 1, Net = 1u << 2,
    Proc = 1u << 3, Gpu = 1u << 4, ForceFullEmit = 1u << 5,
};
constexpr DirtyBit operator|(DirtyBit a, DirtyBit b);
constexpr DirtyBit operator&(DirtyBit a, DirtyBit b);
constexpr bool has_bit(DirtyBit mask, DirtyBit bit);
constexpr DirtyBit DirtyAll = Cpu | Mem | Net | Proc | Gpu;  // no ForceFullEmit

struct PendingDirty {
    void mark(DirtyBit bits);    // fetch_or, release
    DirtyBit take();             // exchange(0), acquire
    DirtyBit peek() const;       // load, acquire
};
```
