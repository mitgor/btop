```cpp
namespace state_tag {
    struct Running {};
    struct Resizing {};
    struct Reloading {};
    struct Sleeping {};
    struct Quitting {};

}

/// Convert runtime AppState enum to compile-time state_tag dispatch.
/// Enables typed overload resolution for on_event(state_tag, event, current).
template<typename Visitor>
decltype(auto) dispatch_state(Global::AppState s, Visitor&& vis) {
    using Global::AppState;
    switch (s) {
        case AppState::Running:   return std::forward<Visitor>(vis)(state_tag::Running{});
        case AppState::Resizing:  return std::forward<Visitor>(vis)(state_tag::Resizing{});
        case AppState::Reloading: return std::forward<Visitor>(vis)(state_tag::Reloading{});
        case AppState::Sleeping:  return std::forward<Visitor>(vis)(state_tag::Sleeping{});
        case AppState::Quitting:  return std::forward<Visitor>(vis)(state_tag::Quitting{});
        case AppState::Error:     return std::forward<Visitor>(vis)(state_tag::Error{});
    }
    __builtin_unreachable();
}
```

1. Add the on_event() overloads. ALL overloads take three parameters: `(state_tag, const event&, AppState current)`. They return `AppState`. Mark them `static`.

**Running state handlers:**
```cpp
//* Typed transition functions — on_event(state_tag, event, current) -> next AppState.
//* Phase 12: coexist with process_signal_event; Plan 02 wires these into the main loop.

static AppState on_event(state_tag::Running, const event::Quit& e, AppState) {
    if (Runner::active) {
        Runner::stopping = true;
        return AppState::Quitting;
    }
    clean_quit(e.exit_code);
    return AppState::Quitting;  // unreachable after clean_quit
}

static AppState on_event(state_tag::Running, const event::Sleep&, AppState) {
    if (Runner::active) {
        Runner::stopping = true;
        return AppState::Sleeping;
    }
    // If runner not active, sleep is handled by process_accumulated_state
    return AppState::Sleeping;
}

static AppState on_event(state_tag::Running, const event::Resume&, AppState) {
    return AppState::Running;  // resume is handled by process_accumulated_state
}

static AppState on_event(state_tag::Running, const event::Resize&, AppState) {
    return AppState::Resizing;
}

static AppState on_event(state_tag::Running, const event::Reload&, AppState) {
    return AppState::Reloading;
}

static AppState on_event(state_tag::Running, const event::TimerTick&, AppState) {
    return AppState::Running;  // timer action handled by main loop
}

static AppState on_event(state_tag::Running, const event::KeyInput&, AppState) {
    return AppState::Running;  // input action handled by main loop
}

static AppState on_event(state_tag::Running, const event::ThreadError&, AppState) {
    return AppState::Error;
}
```

**Catch-all for all unhandled (state, event) combinations:**
```cpp
// Catch-all: unhandled (state, event) pairs preserve current state (no-op).
// This covers: Quitting ignoring all non-ThreadError events, Error ignoring everything,
// Resizing/Reloading/Sleeping ignoring signal events (handled by process_accumulated_state).
static AppState on_event(const auto&, const auto&, AppState current) {
    return current;
}
```

**Note on catch-all:** The `auto` catch-all handles ALL combinations not matched by the specific Running-state overloads above. This means:
- Quitting + any event = stays Quitting (via catch-all)
- Error + any event = stays Error (via catch-all)
- Resizing/Reloading/Sleeping + any signal event = stays in current state (handled later by accumulated state processing)

This is correct for Phase 12. Phase 13 can add specific overloads for other states if needed.

2. Define `dispatch_event()` (the declaration is in btop_events.hpp from Task 1):
```cpp
/// Dispatch a single event against current state via state_tag + std::visit.
AppState dispatch_event(AppState current, const AppEvent& ev) {
    return dispatch_state(current, [&](auto tag) {
        return std::visit([&](const auto& event) {
            return on_event(tag, event, current);
        }, ev);
    });
}
```

Place this right after the on_event overloads and before the signal handler function.

**CRITICAL ORDERING:** The on_event overloads MUST be defined BEFORE dispatch_event in the file, because dispatch_event calls on_event via template instantiation (std::visit + lambda). The catch-all `auto` overload must be the LAST on_event defined, after all specific overloads.

**IMPORTANT:**
- Do NOT modify or remove the existing `process_signal_event` overloads. They remain in use until Plan 02 rewires the main loop.
- Do NOT modify the signal handler or main loop. Plan 02 handles that.
- The on_event functions for Running state that involve side effects (clean_quit, Runner::stopping) have the side effects inlined. This is intentional -- they match the current process_signal_event behavior. However, for TimerTick and KeyInput, the actions are NOT in on_event() -- they remain in the main loop (Runner::run, Input::process). on_event() only returns the state transition; the main loop acts on it.
- The Sleep handler sets Sleeping state and optionally stops the runner, but does NOT call _sleep() -- that is deferred to process_accumulated_state() (Plan 02).
