inline ssize_t read_proc_file(const char* path, char* buf, size_t buf_size) {
    int fd = ::open(path, O_RDONLY | O_CLOEXEC);
    if (fd < 0) return -1;
    ssize_t total = 0;
    while (static_cast<size_t>(total) < buf_size - 1) {
        ssize_t n = ::read(fd, buf + total, buf_size - 1 - total);
        if (n <= 0) break;
        total += n;
    }
    ::close(fd);
    buf[total] = '\0';
    return total;
}
