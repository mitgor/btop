    2. In `src/btop_tools.cpp`, implement `write_stdout()`:
       ```cpp
       void Tools::write_stdout(const char* data, size_t len) {
           while (len > 0) {
               ssize_t n = ::write(STDOUT_FILENO, data, len);
               if (n < 0) {
                   if (errno == EINTR) continue;
                   return;  // Terminal gone or error
               }
               data += n;
               len -= static_cast<size_t>(n);
           }
       }

       void Tools::write_stdout(const std::string& s) {
           write_stdout(s.data(), s.size());
       }
       ```
