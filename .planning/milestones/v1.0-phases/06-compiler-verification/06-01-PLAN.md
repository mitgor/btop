    option(BTOP_PGO_GENERATE "Build with PGO instrumentation (generates profile data)" OFF)
    set(BTOP_PGO_USE "" CACHE FILEPATH "Path to PGO profile data for optimized build")

    if(BTOP_PGO_GENERATE AND BTOP_PGO_USE)
      message(FATAL_ERROR "Cannot set both BTOP_PGO_GENERATE and BTOP_PGO_USE")
    endif()

    # Sanitizer support

    Then, AFTER the `if(BTOP_LTO) ... endif()` block (after line 69), add the PGO and sanitizer flag application:

    ```cmake
    # Apply PGO flags to both targets
    if(BTOP_PGO_GENERATE)
      message(STATUS "PGO: Instrumentation build (profile generation)")
      target_compile_options(libbtop PUBLIC -fprofile-generate)
      target_link_options(libbtop PUBLIC -fprofile-generate)
      target_compile_options(btop PUBLIC -fprofile-generate)
      target_link_options(btop PUBLIC -fprofile-generate)
    endif()

    if(BTOP_PGO_USE)
      if(NOT EXISTS "${BTOP_PGO_USE}")
        message(FATAL_ERROR "PGO profile not found: ${BTOP_PGO_USE}")
      endif()
      message(STATUS "PGO: Optimized build using profile: ${BTOP_PGO_USE}")
      target_compile_options(libbtop PUBLIC -fprofile-use=${BTOP_PGO_USE})
      target_link_options(libbtop PUBLIC -fprofile-use=${BTOP_PGO_USE})
      target_compile_options(btop PUBLIC -fprofile-use=${BTOP_PGO_USE})
      target_link_options(btop PUBLIC -fprofile-use=${BTOP_PGO_USE})
    endif()

    # Apply sanitizer flags
    if(BTOP_SANITIZER)
      message(STATUS "Sanitizer enabled: ${BTOP_SANITIZER}")
      target_compile_options(libbtop PUBLIC
        -fsanitize=${BTOP_SANITIZER}
        -fno-omit-frame-pointer
        -g
      )
      target_link_options(libbtop PUBLIC -fsanitize=${BTOP_SANITIZER})
      target_compile_options(btop PUBLIC
        -fsanitize=${BTOP_SANITIZER}
        -fno-omit-frame-pointer
        -g
      )
      target_link_options(btop PUBLIC -fsanitize=${BTOP_SANITIZER})
    endif()
    ```

    **Part B -- PGO build script (scripts/pgo-build.sh):**

    Create `scripts/` directory if it does not exist. Create `scripts/pgo-build.sh` with executable permission:

    ```bash
    #!/usr/bin/env bash
    # PGO (Profile-Guided Optimization) build script for btop
    # Usage: ./scripts/pgo-build.sh [Release|RelWithDebInfo]
    #
    # This script automates the full PGO workflow:
    # 1. Build instrumented binary (profile generation)
    # 2. Run training workload (btop --benchmark 50)
    # 3. Merge profiles (Clang: llvm-profdata; GCC: automatic)
    # 4. Rebuild with profile data (profile use)
    #
    # The final optimized binary is in build-pgo-use/btop
    set -euo pipefail

    BUILD_TYPE="${1:-Release}"
    CXX_COMPILER="${CXX:-c++}"

    echo "=== PGO Build ==="
    echo "Build type: $BUILD_TYPE"
    echo "Compiler: $CXX_COMPILER"

    # Step 1: Build instrumented binary
    echo ""
    echo "--- Step 1: Building instrumented binary ---"
    cmake -B build-pgo-gen -G Ninja \
      -DCMAKE_BUILD_TYPE="$BUILD_TYPE" \
      -DCMAKE_CXX_COMPILER="$CXX_COMPILER" \
      -DBTOP_PGO_GENERATE=ON \
      -DBTOP_LTO=ON
    cmake --build build-pgo-gen

    # Step 2: Run training workload
    echo ""
    echo "--- Step 2: Running training workload (btop --benchmark 50) ---"
    ./build-pgo-gen/btop --benchmark 50

    # Step 3: Merge profiles
    echo ""
    echo "--- Step 3: Merging profiles ---"
    IS_CLANG=false
    if "$CXX_COMPILER" --version 2>&1 | grep -qi clang; then
      IS_CLANG=true
    fi

    if [ "$IS_CLANG" = true ]; then
      # Clang: merge .profraw files into .profdata
      PROFRAW_FILES=$(find build-pgo-gen -name "*.profraw" 2>/dev/null || find . -name "default_*.profraw" 2>/dev/null)
      if [ -z "$PROFRAW_FILES" ]; then
        # Clang may write to current directory instead of build directory
        PROFRAW_FILES=$(find . -maxdepth 1 -name "*.profraw" 2>/dev/null)
      fi
      if [ -z "$PROFRAW_FILES" ]; then
        echo "ERROR: No .profraw files found. Profile generation may have failed."
        exit 1
      fi
      # shellcheck disable=SC2086
      llvm-profdata merge -output=btop.profdata $PROFRAW_FILES
      PROFILE_PATH="$(pwd)/btop.profdata"
      echo "Clang profile merged: $PROFILE_PATH"
    else
      # GCC: .gcda files are written alongside .gcno files in the build directory
      PROFILE_PATH="$(pwd)/build-pgo-gen"
      GCDA_COUNT=$(find "$PROFILE_PATH" -name "*.gcda" 2>/dev/null | wc -l)
      if [ "$GCDA_COUNT" -eq 0 ]; then
        echo "ERROR: No .gcda files found in $PROFILE_PATH. Profile generation may have failed."
        exit 1
      fi
      echo "GCC profiles: $GCDA_COUNT .gcda files in $PROFILE_PATH"
    fi

    # Step 4: Rebuild with profile data
    echo ""
    echo "--- Step 4: Building optimized binary with PGO ---"
    cmake -B build-pgo-use -G Ninja \
      -DCMAKE_BUILD_TYPE="$BUILD_TYPE" \
      -DCMAKE_CXX_COMPILER="$CXX_COMPILER" \
      -DBTOP_PGO_USE="$PROFILE_PATH" \
      -DBTOP_LTO=ON
    cmake --build build-pgo-use

    echo ""
    echo "=== PGO Build Complete ==="
    echo "Optimized binary: build-pgo-use/btop"
    echo ""
    echo "To benchmark PGO improvement, install hyperfine and run:"
    echo "  hyperfine --warmup 3 --min-runs 10 \\"
    echo "    './build-pgo-gen/btop --benchmark 50' \\"
    echo "    './build-pgo-use/btop --benchmark 50'"
    ```

    Add a new `sanitizer` job to `.github/workflows/cmake-linux.yml` AFTER the existing `build` job. This job runs ASan+UBSan and TSan sweeps using a matrix strategy.

    Add this job block after the existing `build` job:

    ```yaml
      sanitizer:
        name: sanitizer-${{ matrix.name }}
        runs-on: ubuntu-24.04
        concurrency:
          group: ${{ github.workflow }}-${{ github.event.pull_request.number || github.ref }}-sanitizer-${{ matrix.name }}
          cancel-in-progress: true
        strategy:
          fail-fast: false
          matrix:
            include:
              - name: asan-ubsan
                sanitizer: "address,undefined"
                env_opts: "ASAN_OPTIONS=detect_leaks=1:halt_on_error=1 UBSAN_OPTIONS=halt_on_error=1:print_stacktrace=1"
              - name: tsan
                sanitizer: "thread"
                env_opts: "TSAN_OPTIONS=halt_on_error=1:second_deadlock_stack=1"
        steps:
          - uses: actions/checkout@v6

          - name: Install Clang 19
            run: wget -qO - https://apt.llvm.org/llvm.sh | sudo bash -s -- 19 all

          - name: Configure
            run: |
              cmake -B build -G Ninja \
                -DCMAKE_BUILD_TYPE=Debug \
                -DCMAKE_CXX_COMPILER=clang++-19 \
                -DCMAKE_CXX_FLAGS="-stdlib=libc++" \
                -DCMAKE_EXE_LINKER_FLAGS="-fuse-ld=lld-19 -rtlib=compiler-rt -unwindlib=libunwind" \
                -DBTOP_SANITIZER="${{ matrix.sanitizer }}" \
                -DBUILD_TESTING=ON

          - name: Build
            run: cmake --build build

          - name: Run tests
            env:
              ASAN_OPTIONS: ${{ matrix.name == 'asan-ubsan' && 'detect_leaks=1:halt_on_error=1' || '' }}
              UBSAN_OPTIONS: ${{ matrix.name == 'asan-ubsan' && 'halt_on_error=1:print_stacktrace=1' || '' }}
              TSAN_OPTIONS: ${{ matrix.name == 'tsan' && 'halt_on_error=1:second_deadlock_stack=1' || '' }}
            run: ctest --test-dir build --output-on-failure

          - name: Run benchmark workload under sanitizer
            env:
              ASAN_OPTIONS: ${{ matrix.name == 'asan-ubsan' && 'detect_leaks=1:halt_on_error=1' || '' }}
              UBSAN_OPTIONS: ${{ matrix.name == 'asan-ubsan' && 'halt_on_error=1:print_stacktrace=1' || '' }}
              TSAN_OPTIONS: ${{ matrix.name == 'tsan' && 'halt_on_error=1:second_deadlock_stack=1' || '' }}
            run: ./build/btop --benchmark 10
    ```

    **Key details:**
    - Use `fail-fast: false` so both sanitizer configurations run even if one fails.
    - Use Clang 19 (matches existing CI matrix) for both sanitizer builds.
    - Use `-DCMAKE_BUILD_TYPE=Debug` for sanitizer builds (avoids optimizations confusing diagnostics).
    - Use `-DBUILD_TESTING=ON` to enable the test suite.
    - Set sanitizer environment variables as separate `env` blocks per step (not the matrix env_opts string, which doesn't work as a single env block in GitHub Actions).
    - Run both `ctest` (unit tests) and `btop --benchmark 10` (exercises runtime code paths including the runner thread, collectors, and draw functions).
    - The benchmark uses 10 cycles (not 50) since sanitizer overhead is 2-15x.
    - Do NOT add the sanitizer job as a dependency of the build job or vice versa -- they run in parallel.
